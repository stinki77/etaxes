// -------- Адаптер: приема както store-обекти, така и "raw" Payload --------
function incomeTypeToCode(t: Income["incomeType"] | undefined): string {
  switch (t) {
    case "employment": return "01";
    case "civil": return "02";
    case "rent": return "03";
    default: return "99";
  }
}
 
function buildDescription(it: Income): string {
  const parts = [
    it.description || "",
    it.payerName ? `Платец: ${it.payerName}` : "",
    it.payerEik ? `ЕИК: ${it.payerEik}` : "",
    it.countryCode ? `Държава: ${it.countryCode}` : "",
    it.docType || it.docNo || it.docDate ? `Док.: ${[it.docType, it.docNo, it.docDate].filter(Boolean).join(" / ")}` : "",
  ].filter(Boolean);
  return parts.join(" | ");
}
 
/**
 * Разширена сигнатура: приема или store-обекти, или опростен/raw payload.
 * - Ако подадеш { taxpayer, payment, year, income }, ще бъде нормализирано 1:1.
 * - Ако подадеш { person, payment, year, incomes }, ще бъде адаптирано от store.
 */
export function generateNapXml(input: any): string {
  const nowIso = new Date().toISOString();
 
  // Нормализиране на година
  const year: number | string = input?.year ?? input?.meta?.year ?? new Date().getFullYear();
 
  // Taxpayer: приеми и person, и taxpayer
  const person = input?.person ?? {};
  const taxpayerIn = input?.taxpayer ?? {};
  const egn = person.egn ?? taxpayerIn.egn ?? "";
  const lnch = person.lnch ?? taxpayerIn.lnch ?? "";
 
  // Payment
  const refundIban = person.refundIban ?? "";
  const paymentIn = input?.payment ?? {};
  const payment = {
    iban: paymentIn.iban ?? refundIban ?? "",
    reason: paymentIn.reason ?? (typeof input?.paymentReason === "string" ? input.paymentReason : `ГДД ${year}`),
  };
 
  // Incomes: приеми както input.incomes, така и input.income
  const rawIncomes = Array.isArray(input?.incomes)
    ? input.incomes
    : (Array.isArray(input?.income) ? input.income : []);
 
  const incomeLines: IncomeLine[] = rawIncomes.map((it: any) => {
    const code = it?.code != null
      ? String(it.code).padStart(2, "0")
      : incomeTypeToCode(it?.incomeType);
    const description = it?.description != null
      ? String(it.description)
      : (it && (it.incomeType != null || it.payerName || it.payerEik || it.docType || it.docNo || it.docDate))
        ? buildDescription(it as Income)
        : "";
    const amount = n2(it?.amount);
    return { code, description, amount };
  });
 
  // Deductions
  const rawDeductions = Array.isArray(input?.deductions) ? input.deductions : [];
  const mappedDeductions = rawDeductions.map((d: any) => ({ name: d?.name, amount: n2(d?.amount) }));
 
  // Totals
  const totalsIncome = n2(input?.totals?.incomesTotal) || incomeLines.reduce((s, r) => s + n2(r.amount), 0);
  const totalsReliefs = n2(input?.totals?.reliefsTotal) || mappedDeductions.reduce((s, r) => s + n2(r.amount), 0);
 
  const payload: Payload = {
    meta: { generatedAt: nowIso, app: "eTaxes", version: "1.0" },
    taxpayer: { egn, lnch },
    payment,
    year,
    income: incomeLines,
    deductions: mappedDeductions,
    totals: { IncomeTotal: totalsIncome, ReliefsTotal: totalsReliefs },
  };
 
  return generateNapXmlRaw(payload);
}
